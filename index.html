async function getHyperUniqueDeviceID() { const canvas = document.createElement('canvas'); const gl = canvas.getContext('webgl'); const debugInfo = gl.getExtension('WEBGL_debug_renderer_info'); // 1. Hardware: የግራፊክስ ካርድ (GPU) const gpu = gl.getParameter(debugInfo.UNMASKED_RENDERER_WEBGL); // 2. Software & User Preferences (ይሄ ነው ተመሳሳይ ስልክ ያላቸውን የሚለየው!) const info = [ gpu, navigator.userAgent, screen.width, screen.height, window.devicePixelRatio, // በጣም አስፈላጊ: ተጠቃሚው Font/Zoom ከቀየረ ይህ ይቀየራል navigator.language, Intl.DateTimeFormat().resolvedOptions().timeZone, navigator.hardwareConcurrency, navigator.deviceMemory, // *** አዲስ ተጨምሯል: የስርዓት ዝርዝሮች *** navigator.platform, // **Canvas Rendering ልዩነት (በእያንዳንዱ ስልክ ግራፊክስ ድራይቨር የሚለያይ)** canvas.toDataURL(), // **ተጠቃሚው በብሮውዘር የከፈታቸው የፊደል መቼቶች** document.fonts ? document.fonts.size : "unknown" ].join('|||'); // 3. SHA-256 Hash const msgUint8 = new TextEncoder().encode(info); const hashBuffer = await crypto.subtle.digest('SHA-256', msgUint8); const hashArray = Array.from(new Uint8Array(hashBuffer)); return "DEVICE_IMEI_SIMULATOR_" + hashArray.map(b => b.toString(16).padStart(2, '0')).join(''); }
